### 1.3 编写语言与集成归属约束

为了确保 ATE 自动化测试系统的健壮性、可维护性以及产线部署的极简性，本模拟器的具体实现需遵循以下约束：

1. **开发语言约束**：
   * **硬性要求**：在自动化测试正式环境的落地中，必须使用 **JavaScript (Node.js)** 编写串口模拟核心，禁止在正式测试流中采用跨进程调用的 Python/C# 脚本作为核心。
   * **设计考虑**：保持全栈技术栈 100% 统一，避免测试调度器与模拟器之间的跨进程通信（IPC）带来的额外延迟、异常捕捉盲区以及资源泄漏风险。
2. **集成归属约束**：
   * **硬性要求**：模拟器代码**严禁独立成新工程项目**，必须直接作为子模块或子服务继承在当前工程项目中，路径划归至 `backend/ate/SensorSimulator.js`。
   * **设计考虑**：测试调度器 `TestManager.js` 需在内存中直接调用模拟器接口（例如 `setSensorValue`）来针对每个测试场景瞬时切换模拟值与注入故障。单工程集中管理有利于版本同步、一键启动部署（通过 `start-ate.bat`）以及自动化集成测试。

## 二、Modbus RTU 协议分析

### 2.1 环控器发送的读取指令格式

根据 `sensoracquire.c` 代码分析，环控器使用 Modbus RTU 功能码 `0x03` 读取保持寄存器：

```
┌────────────┬────────────┬────────────┬────────────┬────────────┬────────────┐
│  从站地址  │  功能码    │  寄存器地址 │  寄存器数量 │  CRC 低    │  CRC 高    │
│  (1 byte)  │  (1 byte)  │  (2 bytes) │  (2 bytes) │  (1 byte)  │  (1 byte)  │
├────────────┼────────────┼────────────┼────────────┼────────────┼────────────┤
│   0x01     │   0x03     │  0x00 0x00 │  0x00 0x02 │    xx      │    xx      │
└────────────┴────────────┴────────────┴────────────┴────────────┴────────────┘
```

### 2.2 传感器从站响应格式

```
┌────────────┬────────────┬────────────┬────────────┬────────────┬────────────┐
│  从站地址  │  功能码    │  字节数    │  数据区    │  CRC 低    │  CRC 高    │
│  (1 byte)  │  (1 byte)  │  (1 byte)  │ (N bytes)  │  (1 byte)  │  (1 byte)  │
├────────────┼────────────┼────────────┼────────────┼────────────┼────────────┤
│   0x01     │   0x03     │   0x04     │ 温度+湿度  │    xx      │    xx      │
└────────────┴────────────┴────────────┴────────────┴────────────┴────────────┘
```

### 2.3 传感器寄存器映射表

根据传感器地址汇总，常见的寄存器映射：

| 传感器类型 | 从站地址 | 寄存器地址 | 数据格式 | 单位 |
| :--- | :---: | :---: | :--- | :--- |
| 室内温度 1-16 | 0x01-0x10 | 0x0000-0x0001 | INT16 (÷10) | ℃ |
| 室内湿度 1-16 | 0x01-0x10 | 0x0002-0x0003 | INT16 (÷10) | %RH |
| 压差传感器 1-4 | 0x21-0x24 | 0x0000-0x0001 | INT16 (÷10) | Pa |
| CO2 传感器 1-8 | 0x31-0x38 | 0x0000-0x0001 | UINT16 | ppm |
| NH3 传感器 1-4 | 0x41-0x44 | 0x0000-0x0001 | UINT16 | ppm |

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
传感器 Modbus RTU 模拟器
用于环控器自动化测试
"""

import time
import struct
import threading
import yaml
import json
from datetime import datetime
from pymodbus.server.sync import StartSerialServer
from pymodbus.device import ModbusDeviceIdentification
from pymodbus.datastore import ModbusSequentialDataBlock, ModbusSlaveContext, ModbusServerContext
from pymodbus.transaction import ModbusRtuFramer
import serial


class SensorSimulator:
    """传感器模拟器类"""

    def __init__(self, config_file="sensor_config.yaml"):
        """初始化模拟器"""
        self.config = self.load_config(config_file)
        self.sensor_data = {}
        self.error_injection = {}
        self.running = False

        # 初始化传感器数据
        self.init_sensor_data()

    def load_config(self, config_file):
        """加载配置文件"""
        default_config = {
            "serial": {
                "port": "COM3",  # USB-RS485 串口号
                "baudrate": 9600,
                "bytesize": 8,
                "parity": "N",
                "stopbits": 1
            },
            "sensors": {
                "indoor_temp": {
                    "type": "temperature",
                    "count": 16,
                    "address_start": 0x01,
                    "register": 0x0000,
                    "default_value": 25.0,
                    "unit": "℃"
                },
                "indoor_humi": {
                    "type": "humidity",
                    "count": 16,
                    "address_start": 0x01,
                    "register": 0x0002,
                    "default_value": 50.0,
                    "unit": "%RH"
                },
                "indoor_dp": {
                    "type": "pressure",
                    "count": 4,
                    "address_start": 0x21,
                    "register": 0x0000,
                    "default_value": 0.0,
                    "unit": "Pa"
                },
                "co2": {
                    "type": "co2",
                    "count": 8,
                    "address_start": 0x31,
                    "register": 0x0000,
                    "default_value": 400,
                    "unit": "ppm"
                }
            }
        }

        try:
            with open(config_file, 'r', encoding='utf-8') as f:
                import yaml
                return yaml.safe_load(f)
        except FileNotFoundError:
            print(f"配置文件 {config_file} 不存在，使用默认配置")
            return default_config

    def init_sensor_data(self):
        """初始化传感器数据"""
        for sensor_type, sensor_config in self.config["sensors"].items():
            count = sensor_config["count"]
            address_start = sensor_config["address_start"]
            default_value = sensor_config["default_value"]

            for i in range(count):
                slave_addr = address_start + i
                self.sensor_data[slave_addr] = {
                    "type": sensor_type,
                    "value": default_value,
                    "error": None  # None=正常, "timeout"=超时, "fixed"=固定值
                }

    def set_sensor_value(self, slave_addr, value):
        """设置传感器值"""
        if slave_addr in self.sensor_data:
            self.sensor_data[slave_addr]["value"] = value
            self.sensor_data[slave_addr]["error"] = None
            print(f"[模拟器] 设置传感器 {slave_addr} 值为 {value}")

    def inject_error(self, slave_addr, error_type):
        """注入错误"""
        if slave_addr in self.sensor_data:
            self.sensor_data[slave_addr]["error"] = error_type
            print(f"[模拟器] 注入传感器 {slave_addr} 错误: {error_type}")

    def clear_error(self, slave_addr):
        """清除错误"""
        if slave_addr in self.sensor_data:
            self.sensor_data[slave_addr]["error"] = None
            print(f"[模拟器] 清除传感器 {slave_addr} 错误")

    def calculate_crc16(self, data):
        """计算 CRC16"""
        crc = 0xFFFF
        for byte in data:
            crc ^= byte
            for _ in range(8):
                if crc & 0x0001:
                    crc = (crc >> 1) ^ 0xA001
                else:
                    crc >>= 1
        return crc

    def float_to_reg(self, value, scale=10):
        """浮点数转寄存器值"""
        return int(value * scale)

    def reg_to_float(self, reg_value, scale=10):
        """寄存器值转浮点数"""
        return reg_value / scale

    def handle_modbus_request(self, request):
        """处理 Modbus 请求"""
        if len(request) < 4:
            return None

        slave_addr = request[0]
        function_code = request[1]

        # 检查从站地址是否存在
        if slave_addr not in self.sensor_data:
            # 返回异常响应：非法从站地址
            return bytes([slave_addr, 0x81, 0x02])

        # 检查是否注入了超时错误
        if self.sensor_data[slave_addr]["error"] == "timeout":
            return None  # 不响应，模拟超时

        if function_code == 0x03:  # 读保持寄存器
            return self.handle_read_holding_registers(request)
        else:
            # 返回异常响应：非法功能码
            return bytes([slave_addr, 0x83, 0x01])

    def handle_read_holding_registers(self, request):
        """处理读保持寄存器请求"""
        slave_addr = request[0]
        register_addr = (request[2] << 8) | request[3]
        register_count = (request[4] << 8) | request[5]

        # 验证 CRC
        received_crc = (request[-1] << 8) | request[-2]
        calculated_crc = self.calculate_crc16(request[:-2])
        if received_crc != calculated_crc:
            return None  # CRC 错误，不响应

        # 获取传感器值
        sensor_info = self.sensor_data[slave_addr]

        # 检查是否注入了固定值错误
        if sensor_info["error"] == "fixed":
            value = sensor_info["value"]
        else:
            value = sensor_info["value"]

        # 构建响应
        response = bytearray()
        response.append(slave_addr)
        response.append(0x03)
        response.append(register_count * 2)  # 字节数

        # 转换为寄存器值
        reg_value = self.float_to_reg(value)
        response.append((reg_value >> 8) & 0xFF)  # 高字节
        response.append(reg_value & 0xFF)  # 低字节

        # 添加 CRC
        crc = self.calculate_crc16(response)
        response.append(crc & 0xFF)
        response.append((crc >> 8) & 0xFF)

        return bytes(response)

    def start_server(self):
        """启动 Modbus 服务器"""
        self.running = True

        # 创建数据存储
        store = ModbusSlaveContext(
            di=None,
            co=None,
            hr=ModbusSequentialDataBlock(0, [0] * 100),
            ir=None
        )
        context = ModbusServerContext(slaves=store, single=True)

        # 配置串口
        port = self.config["serial"]["port"]
        baudrate = self.config["serial"]["baudrate"]

        print(f"[模拟器] 启动 Modbus RTU 服务器")
        print(f"[模拟器] 串口: {port}, 波特率: {baudrate}")
        print(f"[模拟器] 等待环控器连接...")

        try:
            # 启动 Modbus 服务器
            StartSerialServer(
                context=context,
                framer=ModbusRtuFramer,
                port=port,
                baudrate=baudrate,
                bytesize=8,
                parity='N',
                stopbits=1,
                timeout=3
            )
        except Exception as e:
            print(f"[模拟器] 启动失败: {e}")
            print(f"[模拟器] 尝试使用原始串口模式...")

            # 备用方案：直接使用串口收发
            self.start_raw_serial_server(port, baudrate)

    def start_raw_serial_server(self, port, baudrate):
        """使用原始串口启动服务器（备用方案）"""
        try:
            ser = serial.Serial(
                port=port,
                baudrate=baudrate,
                bytesize=8,
                parity='N',
                stopbits=1,
                timeout=1
            )
            print(f"[模拟器] 串口 {port} 打开成功")

            while self.running:
                # 读取请求
                if ser.in_waiting > 0:
                    request = ser.read(ser.in_waiting)
                    print(f"[模拟器] 收到请求: {request.hex()}")

                    # 处理请求
                    response = self.handle_modbus_request(request)
                    if response:
                        print(f"[模拟器] 发送响应: {response.hex()}")
                        ser.write(response)

                # 更新传感器值（可选：模拟缓慢变化）
                self.update_sensor_values()

        except Exception as e:
            print(f"[模拟器] 串口错误: {e}")
        finally:
            if 'ser' in locals():
                ser.close()

    def update_sensor_values(self):
        """更新传感器值（可选：模拟环境变化）"""
        import random
        for slave_addr, sensor_info in self.sensor_data.items():
            if sensor_info["error"] is None:
                # 模拟小幅波动
                if sensor_info["type"] == "temperature":
                    sensor_info["value"] += random.uniform(-0.1, 0.1)
                elif sensor_info["type"] == "humidity":
                    sensor_info["value"] += random.uniform(-0.5, 0.5)
                    sensor_info["value"] = max(0, min(100, sensor_info["value"]))

    def stop(self):
        """停止模拟器"""
        self.running = False
        print("[模拟器] 已停止")


class SensorSimulatorCLI:
    """传感器模拟器命令行界面"""

    def __init__(self):
        self.simulator = SensorSimulator()

    def run(self):
        """运行 CLI"""
        print("=" * 60)
        print("          环控器传感器 Modbus 模拟器")
        print("=" * 60)
        print()
        print("功能说明：")
        print("  1. 模拟 Modbus RTU 从站响应环控器读取指令")
        print("  2. 支持设置传感器值、注入错误、模拟故障")
        print("  3. 用于自动化测试和调试")
        print()
        print("硬件连接：")
        print("  环控器 RS485 A+ → USB-RS485 A+")
        print("  环控器 RS485 B- → USB-RS485 B-")
        print("  环控器 GND → USB-RS485 GND")
        print()

        # 启动模拟器
        self.simulator.start_server()

    def interactive_mode(self):
        """交互模式"""
        while True:
            print()
            print("可用命令：")
            print("  set <地址> <值>     - 设置传感器值")
            print("  error <地址> <类型> - 注入错误 (timeout/fixed)")
            print("  clear <地址>        - 清除错误")
            print("  show                - 显示当前传感器值")
            print("  quit                - 退出")
            print()

            cmd = input("请输入命令: ").strip().split()

            if not cmd:
                continue

            if cmd[0] == "quit":
                self.simulator.stop()
                break
            elif cmd[0] == "set" and len(cmd) == 3:
                try:
                    addr = int(cmd[1])
                    value = float(cmd[2])
                    self.simulator.set_sensor_value(addr, value)
                except ValueError:
                    print("参数错误")
            elif cmd[0] == "error" and len(cmd) == 3:
                try:
                    addr = int(cmd[1])
                    error_type = cmd[2]
                    self.simulator.inject_error(addr, error_type)
                except ValueError:
                    print("参数错误")
            elif cmd[0] == "clear" and len(cmd) == 2:
                try:
                    addr = int(cmd[1])
                    self.simulator.clear_error(addr)
                except ValueError:
                    print("参数错误")
            elif cmd[0] == "show":
                self.show_sensor_values()
            else:
                print("未知命令")

    def show_sensor_values(self):
        """显示传感器值"""
        print()
        print("当前传感器值：")
        print("-" * 40)
        for addr, info in sorted(self.simulator.sensor_data.items()):
            error_str = f" [ERROR: {info['error']}]" if info['error'] else ""
            print(f"  地址 0x{addr:02X}: {info['value']:.1f}{info['unit']}{error_str}")
        print("-" * 40)


def main():
    """主函数"""
    import sys

    cli = SensorSimulatorCLI()

    if len(sys.argv) > 1 and sys.argv[1] == "--interactive":
        # 交互模式
        cli.interactive_mode()
    else:
        # 服务器模式
        cli.run()


if __name__ == "__main__":
    main()
```

```python
传感器模拟器自动化测试脚本
测试场景：验证环控器传感器采集和异常处理逻辑
```

### 6.1 C# Modbus 从站核心代码

```csharp
using System;
using System.IO.Ports;
using System.Threading;

namespace SensorSimulator
{
    class ModbusSlave
    {
        private SerialPort serialPort;
        private ushort[] holdingRegisters;
        private byte slaveAddress;

        public ModbusSlave(string portName, int baudRate, byte slaveAddr)
        {
            slaveAddress = slaveAddr;
            holdingRegisters = new ushort[100];

            serialPort = new SerialPort(portName, baudRate, Parity.None, 8, StopBits.One);
            serialPort.ReadTimeout = 1000;
            serialPort.WriteTimeout = 1000;
        }

        public void Start()
        {
            serialPort.Open();
            Console.WriteLine($"[Modbus从站] 已启动，监听串口 {serialPort.PortName}");

            while (true)
        {
                try
                {
                    // 读取请求
                    byte[] request = ReadRequest();
                    if (request != null)
                    {
                        // 处理请求
                        byte[] response = ProcessRequest(request);
                        if (response != null)
                        {
                            // 发送响应
                            serialPort.Write(response, 0, response.Length);
                        }
                    }
                }
                catch (TimeoutException)
                {
                    // 读取超时，继续等待
                }
            }
        }

        private byte[] ReadRequest()
        {
            if (serialPort.BytesToRead >= 4)
            {
                byte[] buffer = new byte[serialPort.BytesToRead];
                serialPort.Read(buffer, 0, buffer.Length);
                return buffer;
            }
            return null;
        }

        private byte[] ProcessRequest(byte[] request)
        {
            // 验证从站地址
            if (request[0] != slaveAddress)
                return null;

            // 验证功能码
            if (request[1] != 0x03)
                return GenerateException(0x01); // 非法功能

            // 解析寄存器地址和数量
            ushort startAddr = (ushort)((request[2] << 8) | request[3]);
            ushort count = (ushort)((request[4] << 8) | request[5]);

            // 验证 CRC
            ushort receivedCrc = (ushort)((request[request.Length - 1] << 8) | request[request.Length - 2]);
            ushort calculatedCrc = CalculateCrc16(request, 0, request.Length - 2);
            if (receivedCrc != calculatedCrc)
                return null; // CRC 错误，不响应

            // 构建响应
            byte[] response = new byte[5 + count * 2];
            response[0] = slaveAddress;
            response[1] = 0x03;
            response[2] = (byte)(count * 2);

            for (int i = 0; i < count; i++)
            {
                ushort regValue = holdingRegisters[startAddr + i];
                response[3 + i * 2] = (byte)(regValue >> 8);
                response[4 + i * 2] = (byte)(regValue & 0xFF);
            }

            // 添加 CRC
            ushort crc = CalculateCrc16(response, 0, response.Length - 2);
            response[response.Length - 2] = (byte)(crc & 0xFF);
            response[response.Length - 1] = (byte)(crc >> 8);

            return response;
        }

        private byte[] GenerateException(byte exceptionCode)
        {
            byte[] response = new byte[5];
            response[0] = slaveAddress;
            response[1] = 0x83; // 异常响应
            response[2] = exceptionCode;

            ushort crc = CalculateCrc16(response, 0, 3);
            response[3] = (byte)(crc & 0xFF);
            response[4] = (byte)(crc >> 8);

            return response;
        }

        private ushort CalculateCrc16(byte[] data, int offset, int count)
        {
            ushort crc = 0xFFFF;
            for (int i = offset; i < offset + count; i++)
            {
                crc ^= data[i];
                for (int j = 0; j < 8; j++)
                {
                    if ((crc & 0x0001) != 0)
                        crc = (ushort)((crc >> 1) ^ 0xA001);
                    else
                        crc >>= 1;
                }
            }
            return crc;
        }

        public void SetRegister(ushort address, ushort value)
        {
            if (address < holdingRegisters.Length)
                holdingRegisters[address] = value;
        }
    }
}
```

## 八、Python/C# 方案缺陷规避（避坑指南）

### 8.1 解决多从站（Multi-slave）支持问题
在 Python/C# 版本中，如果环控器向不同从站地址（例如温度 `0x01` ~ `0x10`，压差 `0x21` 等）发送请求，而模拟器只配置了 `single=True`，从站将对非 `0x01` 端口的数据包不作响应。
**规避方法**：
- **Python**：在初始化 `ModbusServerContext` 时，设置 `single=False`，并将每一个所模拟的从站地址通过字典形式注入：
  ```python
  slaves = {}
  for i in range(1, 17): # 模拟 16 个温度从站
      store = ModbusSlaveContext(hr=ModbusSequentialDataBlock(0, [0] * 100))
      slaves[i] = store
  context = ModbusServerContext(slaves=slaves, single=False)
  ```
- **C#**：实例化 `ModbusSlave` 时需要为每个需要响应的 `slaveId` 创建独立的监听器上下文。

### 8.2 避免串口残帧截断导致 CRC 校验失败
在原始串口解析（备用方案）中，如果使用 `ser.read(ser.in_waiting)`，高频或串口数据到达中途执行此命令将读到不完整的 Modbus 帧数据。
**规避方法**：
- 在物理接收端，必须设定字符静默检测定时器，例如 20ms 以上无数据流时才判定断包，或者使用固定长度等待阻塞读取。

## 九、Node.js JavaScript 原生方案（推荐）

为与测试协调器 `TestManager.js` 的技术栈保持 100% 统一，且免去跨进程调用的额外负担，推荐直接在 Node.js 中使用原生 JavaScript 实现串口模拟。

### 9.1 原生从站核心代码 (`backend/ate/SensorSimulator.js`)

```javascript
/**
 * backend/ate/SensorSimulator.js
 * 传感器 Modbus RTU 原生模拟器
 */
const { SerialPort } = require('serialport');

// CRC16 计算辅助
function calculateCRC(buffer) {
  let crc = 0xFFFF;
  for (let i = 0; i < buffer.length; i++) {
    crc ^= buffer[i];
    for (let j = 0; j < 8; j++) {
      if (crc & 0x0001) {
        crc = (crc >> 1) ^ 0xA001;
      } else {
        crc >>= 1;
      }
    }
  }
  return crc;
}

class SensorSimulator {
  constructor() {
    this.port = null;
    this.sensorData = {}; // 存储格式: { [slaveAddr]: { temp: 25.0, humi: 50.0, error: null } }
    this.running = false;
  }

  /**
   * 启动模拟服务
   */
  start(portName, baudRate = 9600) {
    console.log(`[SensorSimulator] 正在打开串口: ${portName}, 波特率: ${baudRate}`);
    
    this.port = new SerialPort({
      path: portName,
      baudRate: baudRate,
      autoOpen: true
    });

    this.running = true;
    let buffer = Buffer.alloc(0);
    let rxTimer = null;

    this.port.on('data', (chunk) => {
      buffer = Buffer.concat([buffer, chunk]);
      
      // 清除之前的定时器，如果在 15ms 内又有数据，说明帧未接收完整
      if (rxTimer) clearTimeout(rxTimer);
      
      rxTimer = setTimeout(() => {
        if (buffer.length >= 8) {
          this.handleRequest(buffer);
        }
        buffer = Buffer.alloc(0);
      }, 15);
    });

    this.port.on('error', (err) => {
      console.error('[SensorSimulator] 串口错误:', err.message);
    });
  }

  /**
   * 解析与回复 Modbus 抄读
   */
  handleRequest(request) {
    const slaveAddr = request[0];
    const funcCode = request[1];

    // 检查从站是否存在
    const sensor = this.sensorData[slaveAddr];
    if (!sensor) return; // 不属于被测试的传感器不回复，避免干扰

    // 检查错误注入
    if (sensor.error === 'timeout') {
      console.log(`[SensorSimulator] 从站 0x${slaveAddr.toString(16)} 注入超时，跳过响应`);
      return;
    }

    // CRC 校验
    const receivedCrc = request.readUInt16LE(request.length - 2);
    const calculatedCrc = calculateCRC(request.subarray(0, request.length - 2));
    if (receivedCrc !== calculatedCrc) {
      console.warn(`[SensorSimulator] 从站 0x${slaveAddr.toString(16)} 收到残缺/CRC校验失败数据包`);
      return;
    }

    if (funcCode === 0x03) { // 读保持寄存器
      const regAddr = request.readUInt16BE(2);
      const regCount = request.readUInt16BE(4);

      // 构造正常响应
      const byteCount = regCount * 2;
      const response = Buffer.alloc(3 + byteCount + 2);
      response[0] = slaveAddr;
      response[1] = 0x03;
      response[2] = byteCount;

      // 模拟寄存器赋值映射
      // 温度 (reg: 0), 湿度 (reg: 2)
      if (regAddr === 0x0000) {
        // 温度写入 (INT16, 放大 10 倍)
        const tempRaw = Math.round(sensor.temp * 10);
        response.writeInt16BE(tempRaw, 3);
        
        if (regCount > 1) {
          // 湿度写入 (INT16, 放大 10 倍)
          const humiRaw = Math.round(sensor.humi * 10);
          response.writeInt16BE(humiRaw, 5);
        }
      } else if (regAddr === 0x0002) {
        const humiRaw = Math.round(sensor.humi * 10);
        response.writeInt16BE(humiRaw, 3);
      }

      // 填充 CRC
      const crc = calculateCRC(response.subarray(0, response.length - 2));
      response.writeUInt16LE(crc, response.length - 2);

      // 发送
      this.port.write(response);
      console.log(`[SensorSimulator] 响应从站 0x${slaveAddr.toString(16)} -> 温度: ${sensor.temp}℃, 湿度: ${sensor.humi}%`);
    }
  }

  /**
   * 动态设置模拟值
   */
  setSensorValue(slaveAddr, temp, humi) {
    this.sensorData[slaveAddr] = {
      temp,
      humi,
      error: null
    };
  }

  /**
   * 动态注入错误
   */
  injectError(slaveAddr, errorType) {
    if (this.sensorData[slaveAddr]) {
      this.sensorData[slaveAddr].error = errorType;
    }
  }

  /**
   * 清除故障
   */
  clearError(slaveAddr) {
    if (this.sensorData[slaveAddr]) {
      this.sensorData[slaveAddr].error = null;
    }
  }

  /**
   * 停止
   */
  stop() {
    this.running = false;
    if (this.port && this.port.isOpen) {
      this.port.close();
    }
    console.log('[SensorSimulator] 服务已关闭');
  }
}

module.exports = SensorSimulator;
```
